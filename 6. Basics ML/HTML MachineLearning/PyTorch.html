Linear regression¶

In [63]:
weigh = torch.tensor(2., requires_grad=True)
bias = torch.tensor(-1., requires_grad=True)
def model(x):
    y = weigh*x + bias
    return y
x = torch.tensor([12])
yhat=model(x)
